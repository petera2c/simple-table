// InfrastructureDemo.ts
import { SimpleTableVanilla } from "simple-table-core";
import type { ColumnDef, CellRenderer } from "simple-table-core";
import { infrastructureData } from "./infrastructure.demo-data"; // 50 sample servers
import "simple-table-core/styles.css";

const cpuRenderer: CellRenderer = ({ row }) => {
  const span = document.createElement("span");
  span.textContent = `${(row.cpuUsage as number).toFixed(1)}%`;
  return span;
};

const headers: ColumnDef[] = [
  { accessor: "serverId", label: "Server ID", type: "string", pinned: "left", width: "1.2fr", sortable: true },
  {
    accessor: "performance",
    label: "Performance Metrics",
    children: [
      // Built-in sparkline column type
      { accessor: "cpuHistory", label: "CPU History", type: "lineAreaChart", width: 150 },
      { accessor: "cpuUsage", label: "CPU %", type: "number", editable: true, width: 120, cellRenderer: cpuRenderer },
      // ...memory, disk, response columns
    ],
  },
  { accessor: "status", label: "Status", type: "enum", width: 130 },
];

export function renderInfrastructureDemo(container: HTMLElement) {
  const table = new SimpleTableVanilla(container, {
    columns: headers,
    rows: infrastructureData,
    columnReordering: true,
    columnResizing: true,
    enableColumnEditor: true,
    selectableCells: true,
    height: "400px",
  });

  // Stream live metrics by calling table.getAPI().updateData(...) on an interval

  return table;
}
