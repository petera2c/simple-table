// ChartsDemo.ts
import { SimpleTableVanilla } from "simple-table-core";
import { chartsConfig } from "./charts.demo-data";
import "simple-table-core/styles.css";

// chartsConfig.headers use built-in chart column types, e.g.
//   { accessor: "trend", type: "lineAreaChart", width: 160 }
//   { accessor: "revenue", type: "barChart", width: 160 }

export function renderChartsDemo(container: HTMLElement) {
  return new SimpleTableVanilla(container, {
    columns: chartsConfig.headers,
    rows: chartsConfig.rows,
    columnReordering: true,
    columnResizing: true,
    selectableCells: true,
    height: "400px",
  });
}
