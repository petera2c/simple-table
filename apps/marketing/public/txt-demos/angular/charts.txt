// charts-demo.component.ts
import { Component } from "@angular/core";
import { SimpleTableComponent } from "@simple-table/angular";
import { chartsConfig } from "./charts.demo-data";
import "@simple-table/angular/styles.css";

// chartsConfig.headers use built-in chart column types, e.g.
//   { accessor: "trend", type: "lineAreaChart", width: 160 }
//   { accessor: "revenue", type: "barChart", width: 160 }

@Component({
  selector: "charts-demo",
  standalone: true,
  imports: [SimpleTableComponent],
  template: `
    <simple-table
      [columns]="headers"
      [rows]="rows"
      [columnReordering]="true"
      [columnResizing]="true"
      [selectableCells]="true"
      height="400px"
    ></simple-table>
  `,
})
export class ChartsDemoComponent {
  readonly rows = chartsConfig.rows;
  readonly headers = chartsConfig.headers;
}
