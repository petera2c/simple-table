// CRMDemo.tsx
import { useState } from "react";
import { SimpleTable } from "@simple-table/react";
import type { ReactColumnDef, CellRendererProps, FooterRendererProps } from "@simple-table/react";
import { crmData } from "./crm.demo-data";
import "@simple-table/react/styles.css";

const headers: ReactColumnDef[] = [
  {
    accessor: "name",
    label: "CONTACT",
    type: "string",
    width: "2fr",
    sortable: true,
    cellRenderer: ({ row }: CellRendererProps) => <strong>{row.name as string}</strong>,
  },
  { accessor: "signal", label: "SIGNAL", type: "string", width: "3fr" },
  { accessor: "aiScore", label: "AI SCORE", type: "number", align: "center" },
  { accessor: "emailStatus", label: "EMAIL", type: "enum", align: "center" },
  // ...list, fit, contact columns
];

const CRMDemo = () => {
  const [rows, setRows] = useState([...crmData]);

  return (
    <SimpleTable
      columns={headers}
      rows={rows}
      enableRowSelection
      enablePagination
      rowsPerPage={100}
      columnReordering
      columnResizing
      customTheme={{ headerHeight: 48, rowHeight: 92 }}
      onCellEdit={({ accessor, newValue, row }) =>
        setRows((prev) => prev.map((r) => (r.id === row.id ? { ...r, [accessor]: newValue } : r)))
      }
      // Pass footerRenderer={(props: FooterRendererProps) => ...} for custom pagination UI
      height="400px"
    />
  );
};

export default CRMDemo;
