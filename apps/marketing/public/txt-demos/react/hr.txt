// HRDemo.tsx
import { useState } from "react";
import { SimpleTable } from "@simple-table/react";
import type { CellChangeProps } from "@simple-table/react";
import { hrConfig } from "./hr.demo-data";
import "@simple-table/react/styles.css";

const HRDemo = () => {
  const [rows, setRows] = useState([...hrConfig.rows]);

  const handleCellEdit = ({ accessor, newValue, row }: CellChangeProps) =>
    setRows((prev) => prev.map((r) => (r.id === row.id ? { ...r, [accessor]: newValue } : r)));

  return (
    <SimpleTable
      columns={hrConfig.headers}
      rows={rows}
      enablePagination
      rowsPerPage={8}
      customTheme={{ rowHeight: 48 }}
      columnReordering
      columnResizing
      selectableCells
      onCellEdit={handleCellEdit}
      height="400px"
    />
  );
};

export default HRDemo;
