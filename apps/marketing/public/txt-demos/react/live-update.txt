// LiveUpdateDemo.tsx
import { useRef, useEffect } from "react";
import { SimpleTable } from "@simple-table/react";
import type { TableAPI } from "@simple-table/react";
import { liveUpdateConfig } from "./live-update.demo-data";
import "@simple-table/react/styles.css";

const LiveUpdateDemo = () => {
  const tableRef = useRef<TableAPI>(null);

  useEffect(() => {
    const id = setInterval(() => {
      const rowIndex = Math.floor(Math.random() * liveUpdateConfig.rows.length);
      const newPrice = +(Math.random() * 100).toFixed(2);
      // Push a single-cell update without re-rendering the whole table
      tableRef.current?.updateData({ accessor: "price", rowIndex, newValue: newPrice });
    }, 500);
    return () => clearInterval(id);
  }, []);

  return (
    <SimpleTable
      ref={tableRef}
      columns={liveUpdateConfig.headers}
      rows={liveUpdateConfig.rows}
      height="400px"
    />
  );
};

export default LiveUpdateDemo;
