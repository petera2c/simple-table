// InfrastructureDemo.tsx
import { useRef } from "react";
import { SimpleTable } from "@simple-table/react";
import type { TableAPI, ReactColumnDef, CellRendererProps } from "@simple-table/react";
import { infrastructureData } from "./infrastructure.demo-data"; // 50 sample servers
import "@simple-table/react/styles.css";

const headers: ReactColumnDef[] = [
  { accessor: "serverId", label: "Server ID", type: "string", pinned: "left", width: "1.2fr", sortable: true },
  {
    accessor: "performance",
    label: "Performance Metrics",
    children: [
      // Built-in sparkline column type
      { accessor: "cpuHistory", label: "CPU History", type: "lineAreaChart", width: 150 },
      {
        accessor: "cpuUsage",
        label: "CPU %",
        type: "number",
        editable: true,
        width: 120,
        cellRenderer: ({ row }: CellRendererProps) => <span>{(row.cpuUsage as number).toFixed(1)}%</span>,
      },
      // ...memory, disk, response columns
    ],
  },
  { accessor: "status", label: "Status", type: "enum", width: 130 },
];

const InfrastructureDemo = () => {
  const tableRef = useRef<TableAPI>(null);

  // Stream live metrics by calling tableRef.current.updateData(...) on an interval

  return (
    <SimpleTable
      ref={tableRef}
      columns={headers}
      rows={infrastructureData}
      columnReordering
      columnResizing
      enableColumnEditor
      selectableCells
      height="400px"
    />
  );
};

export default InfrastructureDemo;
