// live-update-demo.component.ts
import { AfterViewInit, Component, OnDestroy, ViewChild } from "@angular/core";
import { SimpleTableComponent } from "@simple-table/angular";
import { liveUpdateConfig } from "./live-update.demo-data";
import "@simple-table/angular/styles.css";

@Component({
  selector: "live-update-demo",
  standalone: true,
  imports: [SimpleTableComponent],
  template: `
    <simple-table
      #simpleTable
      [defaultHeaders]="headers"
      [rows]="rows"
      height="400px"
    ></simple-table>
  `,
})
export class LiveUpdateDemoComponent implements AfterViewInit, OnDestroy {
  @ViewChild("simpleTable") tableRef!: SimpleTableComponent;
  readonly rows = liveUpdateConfig.rows;
  readonly headers = liveUpdateConfig.headers;
  private id?: ReturnType<typeof setInterval>;

  ngAfterViewInit(): void {
    this.id = setInterval(() => {
      const rowIndex = Math.floor(Math.random() * this.rows.length);
      const newPrice = +(Math.random() * 100).toFixed(2);
      // Push a single-cell update without re-rendering the whole table
      this.tableRef.getAPI()?.updateData({ accessor: "price", rowIndex, newValue: newPrice });
    }, 500);
  }

  ngOnDestroy(): void {
    clearInterval(this.id);
  }
}
