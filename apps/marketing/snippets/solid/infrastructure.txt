// InfrastructureDemo.tsx
import { SimpleTable } from "@simple-table/solid";
import type { TableAPI, SolidHeaderObject } from "@simple-table/solid";
import { infrastructureData } from "./infrastructure.demo-data"; // 50 sample servers
import "@simple-table/solid/styles.css";

const headers: SolidHeaderObject[] = [
  { accessor: "serverId", label: "Server ID", type: "string", pinned: "left", width: "1.2fr", isSortable: true },
  {
    accessor: "performance",
    label: "Performance Metrics",
    children: [
      // Built-in sparkline column type
      { accessor: "cpuHistory", label: "CPU History", type: "lineAreaChart", width: 150 },
      {
        accessor: "cpuUsage",
        label: "CPU %",
        type: "number",
        isEditable: true,
        width: 120,
        cellRenderer: ({ row }) => <span>{(row.cpuUsage as number).toFixed(1)}%</span>,
      },
      // ...memory, disk, response columns
    ],
  },
  { accessor: "status", label: "Status", type: "enum", width: 130 },
];

export default function InfrastructureDemo() {
  let tableRef: TableAPI | undefined;

  // Stream live metrics by calling tableRef.updateData(...) on an interval

  return (
    <SimpleTable
      ref={(api) => (tableRef = api)}
      defaultHeaders={headers}
      rows={infrastructureData}
      columnReordering
      columnResizing
      editColumns
      selectableCells
      height="400px"
    />
  );
}
