// CRMDemo.tsx
import { createSignal } from "solid-js";
import { SimpleTable } from "@simple-table/solid";
import type { SolidHeaderObject } from "@simple-table/solid";
import { crmData } from "./crm.demo-data";
import "@simple-table/solid/styles.css";

const headers: SolidHeaderObject[] = [
  {
    accessor: "name",
    label: "CONTACT",
    type: "string",
    width: "2fr",
    sortable: true,
    cellRenderer: ({ row }) => <strong>{row.name as string}</strong>,
  },
  { accessor: "signal", label: "SIGNAL", type: "string", width: "3fr" },
  { accessor: "aiScore", label: "AI SCORE", type: "number", align: "center" },
  { accessor: "emailStatus", label: "EMAIL", type: "enum", align: "center" },
  // ...list, fit, contact columns
];

export default function CRMDemo() {
  const [rows] = createSignal([...crmData]);

  return (
    <SimpleTable
      defaultHeaders={headers}
      rows={rows()}
      enableRowSelection
      shouldPaginate
      rowsPerPage={100}
      columnReordering
      columnResizing
      customTheme={{ headerHeight: 48, rowHeight: 92 }}
      height="400px"
    />
  );
}
