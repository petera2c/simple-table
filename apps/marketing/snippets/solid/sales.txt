// SalesDemo.tsx
import { createSignal } from "solid-js";
import { SimpleTable } from "@simple-table/solid";
import type { FooterRendererProps } from "@simple-table/solid";
import { generateSalesData, type SalesRow } from "./sales.demo-data";
import { SALES_HEADERS } from "./sales-headers";
import "@simple-table/solid/styles.css";

export default function SalesDemo() {
  const [rows] = createSignal<SalesRow[]>(generateSalesData(240));

  return (
    <SimpleTable
      defaultHeaders={SALES_HEADERS}
      rows={rows()}
      shouldPaginate
      rowsPerPage={40}
      footerPosition="top"
      footerRenderer={(props: FooterRendererProps) => (
        <span>Page {props.currentPage} of {props.totalPages}</span>
      )}
      initialSortColumn="dealValue"
      initialSortDirection="desc"
      autoExpandColumns
      columnReordering
      columnResizing
      editColumns
      selectableCells
      height="70dvh"
    />
  );
}
