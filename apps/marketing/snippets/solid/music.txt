// MusicDemo.tsx
import { SimpleTable } from "@simple-table/solid";
import type { SolidHeaderObject, CellRendererProps } from "@simple-table/solid";
import { musicData } from "./music.demo-data";
import "@simple-table/solid/styles.css";

const headers: SolidHeaderObject[] = [
  { accessor: "rank", label: "#", type: "number", width: 60, pinned: "left", align: "center" },
  {
    accessor: "artistName",
    label: "Artist",
    type: "string",
    width: 330,
    pinned: "left",
    cellRenderer: ({ row }: CellRendererProps) => <strong>{row.artistName as string}</strong>,
  },
  {
    accessor: "followersGroup",
    label: "Followers",
    width: 700,
    collapsible: true,
    children: [
      { accessor: "followers", label: "Total", type: "number", width: 180, showWhen: "always" },
      { accessor: "followers7DayGrowth", label: "7-Day", type: "number", width: 160, showWhen: "parentExpanded" },
      // ...28-day, 60-day growth columns
    ],
  },
  // ...popularity, playlist reach, monthly listeners groups
];

export default function MusicDemo() {
  return (
    <SimpleTable
      defaultHeaders={headers}
      rows={musicData}
      columnReordering
      columnResizing
      customTheme={{ headerHeight: 30, rowHeight: 85 }}
      selectableCells
      height="400px"
    />
  );
}
