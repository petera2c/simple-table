// HRDemo.tsx
import { createSignal } from "solid-js";
import { SimpleTable } from "@simple-table/solid";
import type { CellChangeProps } from "@simple-table/solid";
import { hrConfig } from "./hr.demo-data";
import "@simple-table/solid/styles.css";

export default function HRDemo() {
  const [rows, setRows] = createSignal([...hrConfig.rows]);

  const handleCellEdit = ({ accessor, newValue, row }: CellChangeProps) =>
    setRows((prev) => prev.map((r) => (r.id === row.id ? { ...r, [accessor]: newValue } : r)));

  return (
    <SimpleTable
      defaultHeaders={hrConfig.headers}
      rows={rows()}
      shouldPaginate
      rowsPerPage={8}
      customTheme={{ rowHeight: 48 }}
      columnReordering
      columnResizing
      selectableCells
      onCellEdit={handleCellEdit}
      height="400px"
    />
  );
}
