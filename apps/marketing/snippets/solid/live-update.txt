// LiveUpdateDemo.tsx
import { onMount, onCleanup } from "solid-js";
import { SimpleTable } from "@simple-table/solid";
import type { TableAPI } from "@simple-table/solid";
import { liveUpdateConfig } from "./live-update.demo-data";
import "@simple-table/solid/styles.css";

export default function LiveUpdateDemo() {
  let tableRef: TableAPI | undefined;

  onMount(() => {
    const id = setInterval(() => {
      const rowIndex = Math.floor(Math.random() * liveUpdateConfig.rows.length);
      const newPrice = +(Math.random() * 100).toFixed(2);
      // Push a single-cell update without re-rendering the whole table
      tableRef?.updateData({ accessor: "price", rowIndex, newValue: newPrice });
    }, 500);
    onCleanup(() => clearInterval(id));
  });

  return (
    <SimpleTable
      ref={(api) => (tableRef = api)}
      defaultHeaders={liveUpdateConfig.headers}
      rows={liveUpdateConfig.rows}
      height="400px"
    />
  );
}
