// LiveUpdateDemo.vue
<template>
  <SimpleTable
    ref="tableRef"
    :columns="liveUpdateConfig.headers"
    :rows="liveUpdateConfig.rows"
    height="400px"
  />
</template>

<script setup lang="ts">
import { ref, onMounted, onUnmounted } from "vue";
import { SimpleTable } from "@simple-table/vue";
import { liveUpdateConfig } from "./live-update.demo-data";
import "@simple-table/vue/styles.css";

const tableRef = ref();
let id: ReturnType<typeof setInterval>;

onMounted(() => {
  id = setInterval(() => {
    const rowIndex = Math.floor(Math.random() * liveUpdateConfig.rows.length);
    const newPrice = +(Math.random() * 100).toFixed(2);
    // Push a single-cell update without re-rendering the whole table
    tableRef.value?.getAPI()?.updateData({ accessor: "price", rowIndex, newValue: newPrice });
  }, 500);
});

onUnmounted(() => clearInterval(id));
</script>
