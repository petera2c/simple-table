// LiveUpdateDemo.svelte
<script lang="ts">
  import { onMount } from "svelte";
  import { SimpleTable } from "@simple-table/svelte";
  import { liveUpdateConfig } from "./live-update.demo-data";
  import "@simple-table/svelte/styles.css";

  let tableRef: any;

  onMount(() => {
    const id = setInterval(() => {
      const rowIndex = Math.floor(Math.random() * liveUpdateConfig.rows.length);
      const newPrice = +(Math.random() * 100).toFixed(2);
      // Push a single-cell update without re-rendering the whole table
      tableRef?.getAPI()?.updateData({ accessor: "price", rowIndex, newValue: newPrice });
    }, 500);
    return () => clearInterval(id);
  });
</script>

<SimpleTable
  bind:this={tableRef}
  defaultHeaders={liveUpdateConfig.headers}
  rows={liveUpdateConfig.rows}
  height="400px"
/>
