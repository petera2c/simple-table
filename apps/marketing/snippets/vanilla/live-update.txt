// LiveUpdateDemo.ts
import { SimpleTableVanilla } from "simple-table-core";
import { liveUpdateConfig } from "./live-update.demo-data";
import "simple-table-core/styles.css";

export function renderLiveUpdateDemo(container: HTMLElement) {
  const table = new SimpleTableVanilla(container, {
    defaultHeaders: liveUpdateConfig.headers,
    rows: liveUpdateConfig.rows,
    height: "400px",
  });

  setInterval(() => {
    const rowIndex = Math.floor(Math.random() * liveUpdateConfig.rows.length);
    const newPrice = +(Math.random() * 100).toFixed(2);
    // Push a single-cell update without re-rendering the whole table
    table.getAPI()?.updateData({ accessor: "price", rowIndex, newValue: newPrice });
  }, 500);

  return table;
}
