// SalesDemo.ts
import { SimpleTableVanilla } from "simple-table-core";
import { generateSalesData } from "./sales.demo-data";
import { SALES_HEADERS } from "./sales-headers";
import "simple-table-core/styles.css";

export function renderSalesDemo(container: HTMLElement) {
  return new SimpleTableVanilla(container, {
    defaultHeaders: SALES_HEADERS,
    rows: generateSalesData(240),
    shouldPaginate: true,
    rowsPerPage: 40,
    footerPosition: "top",
    initialSortColumn: "dealValue",
    initialSortDirection: "desc",
    autoExpandColumns: true,
    columnReordering: true,
    columnResizing: true,
    editColumns: true,
    selectableCells: true,
    height: "70dvh",
  });
}
