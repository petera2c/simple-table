// HRDemo.ts
import { SimpleTableVanilla } from "simple-table-core";
import { hrConfig } from "./hr.demo-data";
import "simple-table-core/styles.css";

export function renderHRDemo(container: HTMLElement) {
  return new SimpleTableVanilla(container, {
    defaultHeaders: hrConfig.headers,
    rows: [...hrConfig.rows],
    shouldPaginate: true,
    rowsPerPage: 8,
    customTheme: { rowHeight: 48 },
    columnReordering: true,
    columnResizing: true,
    selectableCells: true,
    height: "400px",
  });
}
