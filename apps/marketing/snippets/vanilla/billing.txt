// BillingDemo.ts
import { SimpleTableVanilla } from "simple-table-core";
import { billingConfig } from "./billing.demo-data";
import "simple-table-core/styles.css";

export function renderBillingDemo(container: HTMLElement) {
  return new SimpleTableVanilla(container, {
    defaultHeaders: billingConfig.headers,
    rows: billingConfig.rows,
    rowGrouping: ["invoices", "charges"],
    initialSortColumn: "amount",
    initialSortDirection: "desc",
    columnReordering: true,
    columnResizing: true,
    editColumns: true,
    selectableCells: true,
    useOddColumnBackground: true,
    height: "400px",
  });
}
