// MusicDemo.ts
import { SimpleTableVanilla } from "simple-table-core";
import type { ColumnDef } from "simple-table-core";
import { musicData } from "./music.demo-data";
import "simple-table-core/styles.css";

const headers: ColumnDef[] = [
  { accessor: "rank", label: "#", type: "number", width: 60, pinned: "left", align: "center" },
  { accessor: "artistName", label: "Artist", type: "string", width: 330, pinned: "left" },
  {
    accessor: "followersGroup",
    label: "Followers",
    width: 700,
    collapsible: true,
    children: [
      { accessor: "followers", label: "Total", type: "number", width: 180, showWhen: "always" },
      { accessor: "followers7DayGrowth", label: "7-Day", type: "number", width: 160, showWhen: "parentExpanded" },
      // ...28-day, 60-day growth columns
    ],
  },
  // ...popularity, playlist reach, monthly listeners groups
];

export function renderMusicDemo(container: HTMLElement) {
  return new SimpleTableVanilla(container, {
    columns: headers,
    rows: musicData,
    columnReordering: true,
    columnResizing: true,
    customTheme: { headerHeight: 30, rowHeight: 85 },
    selectableCells: true,
    height: "400px",
  });
}
