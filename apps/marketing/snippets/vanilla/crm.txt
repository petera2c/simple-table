// CRMDemo.ts
import { SimpleTableVanilla } from "simple-table-core";
import type { ColumnDef, CellRenderer } from "simple-table-core";
import { crmData } from "./crm.demo-data";
import "simple-table-core/styles.css";

const nameRenderer: CellRenderer = ({ row }) => {
  const el = document.createElement("strong");
  el.textContent = String(row.name);
  return el;
};

const headers: ColumnDef[] = [
  { accessor: "name", label: "CONTACT", type: "string", width: "2fr", sortable: true, cellRenderer: nameRenderer },
  { accessor: "signal", label: "SIGNAL", type: "string", width: "3fr" },
  { accessor: "aiScore", label: "AI SCORE", type: "number", align: "center" },
  { accessor: "emailStatus", label: "EMAIL", type: "enum", align: "center" },
  // ...list, fit, contact columns
];

export function renderCRMDemo(container: HTMLElement) {
  return new SimpleTableVanilla(container, {
    columns: headers,
    rows: [...crmData],
    enableRowSelection: true,
    enablePagination: true,
    rowsPerPage: 100,
    columnReordering: true,
    columnResizing: true,
    customTheme: { headerHeight: 48, rowHeight: 92 },
    height: "400px",
  });
}
