// ManufacturingDemo.ts
import { SimpleTableVanilla } from "simple-table-core";
import { manufacturingConfig } from "./manufacturing.demo-data";
import "simple-table-core/styles.css";

export function renderManufacturingDemo(container: HTMLElement) {
  return new SimpleTableVanilla(container, {
    columns: manufacturingConfig.headers,
    rows: manufacturingConfig.rows,
    rowGrouping: ["stations"],
    columnReordering: true,
    columnResizing: true,
    selectableCells: true,
    height: "400px",
  });
}
