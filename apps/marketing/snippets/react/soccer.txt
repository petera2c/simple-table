// SportsDemo.tsx
import { SimpleTable } from "@simple-table/react";
import type { ReactColumnDef, CellRendererProps } from "@simple-table/react";
import { generateSportsData } from "./sports-data";

const data = generateSportsData(600);

const RatingPill = ({ row }: CellRendererProps) => {
  const rating = row.rating as number;
  const bg = rating >= 7.5 ? "#dcfce7" : rating >= 7 ? "#dbeafe" : "#fef3c7";
  const text = rating >= 7.5 ? "#16a34a" : rating >= 7 ? "#2563eb" : "#d97706";
  return (
    <span style={{ background: bg, color: text, padding: "3px 8px", borderRadius: 6, fontWeight: 700 }}>
      {rating.toFixed(2)}
    </span>
  );
};

// Every column uses width: "auto" so it sizes to its content automatically
const headers: ReactColumnDef[] = [
  { accessor: "rank", label: "#", type: "number", width: "auto", pinned: "left", align: "center" },
  {
    accessor: "player",
    label: "Player",
    width: "auto",
    pinned: "left",
    valueGetter: ({ row }) => row.name as string,
    cellRenderer: ({ row }: CellRendererProps) => (
      <strong>
        {row.nationFlag as string} {row.name as string} — {row.clubShort as string}
      </strong>
    ),
  },
  {
    accessor: "position",
    label: "Pos",
    width: "auto",
    align: "center",
    type: "enum",
    filterable: true,
    enumOptions: [
      { label: "Goalkeeper", value: "GK" },
      { label: "Defender", value: "DEF" },
      { label: "Midfielder", value: "MID" },
      { label: "Forward", value: "FWD" },
    ],
  },
  { accessor: "rating", label: "Rating", width: "auto", align: "center", type: "number", cellRenderer: RatingPill },
  // Per-row form sparkline from last 8 match ratings
  { accessor: "formHistory", label: "Form", width: "auto", type: "lineAreaChart" },
  {
    accessor: "attack",
    label: "Attack",
    width: "auto",
    collapsible: true,
    children: [
      { accessor: "goals", label: "Goals", width: "auto", align: "center", type: "number", showWhen: "always" },
      { accessor: "assists", label: "Assists", width: "auto", align: "center", type: "number", showWhen: "parentExpanded" },
      { accessor: "xG", label: "xG", width: "auto", align: "center", type: "number", showWhen: "parentExpanded" },
      // ...xA, shots
    ],
  },
  // ...Passing and Defense collapsible groups
];

const SportsDemo = () => (
  <SimpleTable
    columns={headers}
    rows={data}
    columnReordering
    columnResizing
    enableColumnEditor
    selectableCells
    initialSortColumn="rating"
    initialSortDirection="desc"
    customTheme={{ headerHeight: 40, rowHeight: 60 }}
    height="70dvh"
  />
);

export default SportsDemo;
