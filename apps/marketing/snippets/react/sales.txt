// SalesDemo.tsx
import { useState } from "react";
import { SimpleTable } from "@simple-table/react";
import type { CellChangeProps, FooterRendererProps } from "@simple-table/react";
import { generateSalesData, type SalesRow } from "./sales.demo-data";
import { SALES_HEADERS } from "./sales-headers";
import "@simple-table/react/styles.css";

const SalesDemo = () => {
  const [rows, setRows] = useState<SalesRow[]>(() => generateSalesData(240));

  const handleCellEdit = ({ accessor, newValue, row }: CellChangeProps) =>
    setRows((prev) => prev.map((r) => (r.id === row.id ? { ...r, [accessor]: newValue } : r)));

  return (
    <SimpleTable
      columns={SALES_HEADERS}
      rows={rows}
      enablePagination
      rowsPerPage={40}
      footerPosition="top"
      footerRenderer={(props: FooterRendererProps) => (
        <span>Page {props.currentPage} of {props.totalPages}</span>
      )}
      initialSortColumn="dealValue"
      initialSortDirection="desc"
      autoExpandColumns
      columnReordering
      columnResizing
      enableColumnEditor
      selectableCells
      onCellEdit={handleCellEdit}
      height="70dvh"
    />
  );
};

export default SalesDemo;
