// CryptoDemo.tsx
import { useRef } from "react";
import { SimpleTable } from "@simple-table/react";
import type { ReactHeaderObject, CellRendererProps, TableAPI } from "@simple-table/react";
import { generateCryptoData } from "./crypto-data";

const data = generateCryptoData(1000);

const ChangeCell =
  (accessor: string) =>
  ({ row }: CellRendererProps) => {
    const value = row[accessor] as number;
    const up = value >= 0;
    return (
      <span style={{ color: up ? "#16a34a" : "#dc2626", fontWeight: 600 }}>
        {up ? "▲" : "▼"} {Math.abs(value).toFixed(2)}%
      </span>
    );
  };

const headers: ReactHeaderObject[] = [
  { accessor: "rank", label: "#", type: "number", width: 56, pinned: "left", align: "center" },
  {
    accessor: "asset",
    label: "Asset",
    width: 240,
    pinned: "left",
    valueGetter: ({ row }) => row.name as string,
    cellRenderer: ({ row }: CellRendererProps) => (
      <strong>
        {row.name as string} <span style={{ opacity: 0.6 }}>{row.symbol as string}</span>
      </strong>
    ),
  },
  {
    accessor: "price",
    label: "Price",
    width: 130,
    align: "right",
    type: "number",
    cellRenderer: ({ row }: CellRendererProps) => `$${(row.price as number).toLocaleString()}`,
  },
  // Per-row sparkline from a 30-point price history array
  { accessor: "priceHistory", label: "Last 30d", width: 160, type: "lineAreaChart" },
  {
    accessor: "performance",
    label: "Performance",
    width: 480,
    collapsible: true,
    children: [
      { accessor: "change24h", label: "24h %", width: 120, align: "right", type: "number", showWhen: "always", cellRenderer: ChangeCell("change24h") },
      { accessor: "change7d", label: "7d %", width: 110, align: "right", type: "number", showWhen: "parentExpanded", cellRenderer: ChangeCell("change7d") },
      // ...1h, 30d
    ],
  },
  // ...Market group: marketCap, volume, circulating-supply bar
];

const CryptoDemo = () => {
  const tableRef = useRef<TableAPI | null>(null);

  // Live "market feed": tick visible rows so prices + the sparkline flash in place
  // (see useCryptoTicker -> tableRef.current.updateData({ accessor, rowIndex, newValue })).

  // External / window scroll: the table has NO height, so it grows to its natural
  // size and the page scroll drives row virtualization. scrollParent="window" lets
  // the whole page scroll; the header auto-pins to the top of the viewport. (Pass
  // an element or () => el to scroll inside a specific container instead.)
  return (
    <SimpleTable
      ref={tableRef}
      defaultHeaders={headers}
      rows={data}
      columnReordering
      columnResizing
      editColumns
      selectableCells
      customTheme={{ headerHeight: 40, rowHeight: 64 }}
      scrollParent="window"
    />
  );
};

export default CryptoDemo;
